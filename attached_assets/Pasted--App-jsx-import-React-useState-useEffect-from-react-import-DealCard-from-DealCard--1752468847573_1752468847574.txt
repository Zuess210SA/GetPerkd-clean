// App.jsx
import React, { useState, useEffect } from 'react';
import DealCard from './DealCard';
import BusinessDashboard from './BusinessDashboard';
import AuthPage from './AuthPage';
import { supabase } from './supabaseClient';

function App() {
  const [view, setView] = useState('student');
  const [session, setSession] = useState(null);
  const [deals, setDeals] = useState([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    supabase.auth.getSession().then(({ data: { session } }) => {
      setSession(session);
    });

    const { data: listener } = supabase.auth.onAuthStateChange((_event, session) => {
      setSession(session);
    });

    return () => {
      listener?.subscription.unsubscribe();
    };
  }, []);

  useEffect(() => {
    if (session) fetchDeals();
  }, [session]);

  const fetchDeals = async () => {
    setLoading(true);
    const { data, error } = await supabase.from('deals').select('*');
    if (!error) setDeals(data);
    setLoading(false);
  };

  const handleNewDeal = async (newDeal) => {
    const { data, error } = await supabase.from('deals').insert([newDeal]);
    if (!error) fetchDeals();
  };

  if (!session) return <AuthPage onAuthSuccess={(session) => setSession(session)} />;

  return (
    <div className="min-h-screen bg-gray-100 p-6">
      <div className="mb-6 text-center">
        <button
          onClick={() => setView('student')}
          className={`mr-4 px-4 py-2 rounded ${view === 'student' ? 'bg-blue-600 text-white' : 'bg-white border'}`}
        >
          Student View
        </button>
        <button
          onClick={() => setView('business')}
          className={`px-4 py-2 rounded ${view === 'business' ? 'bg-blue-600 text-white' : 'bg-white border'}`}
        >
          Business Dashboard
        </button>
      </div>

      {view === 'student' ? (
        <>
          <h1 className="text-3xl font-bold text-center mb-6">ğŸ Get Perkd: Student Deals</h1>
          {loading ? (
            <p className="text-center">Loading deals...</p>
          ) : deals.length === 0 ? (
            <p className="text-center text-gray-600">No deals available yet.</p>
          ) : (
            deals.map((deal) => (
              <DealCard
                key={deal.id}
                deal={deal}
                redeemedInfo={null} // hook up redemption tracking later
                onRedeem={() => {}} // hook up redeem later
              />
            ))
          )}
        </>
      ) : (
        <BusinessDashboard onNewDeal={handleNewDeal} />
      )}
    </div>
  );
}

export default App;
