// StaffDashboard.jsx
import React, { useEffect, useState } from 'react';
import { supabase } from './supabaseClient';

function StaffDashboard({ staff }) {
  const [deals, setDeals] = useState([]);
  const [loading, setLoading] = useState(true);
  const [message, setMessage] = useState('');

  useEffect(() => {
    const fetchDeals = async () => {
      const { data, error } = await supabase
        .from('deals')
        .select('*')
        .eq('merchant_id', staff.merchant_id);

      if (error) {
        console.error(error);
      } else {
        setDeals(data || []);
      }

      setLoading(false);
    };

    fetchDeals();
  }, [staff]);

  const handleRedeem = async (dealId) => {
    const { error } = await supabase.from('redemptions').insert([
      {
        deal_id: dealId,
        redeemed_by: staff.email,
        redeemed_at: new Date().toISOString(),
      },
    ]);

    if (error) {
      console.error('Redemption failed:', error);
      setMessage('âŒ Failed to redeem.');
    } else {
      setMessage('âœ… Deal redeemed successfully! ğŸ‰ğŸº');
    }
  };

  if (loading) return <p className="text-center mt-12">Loading deals for redemption...</p>;

  return (
    <div className="max-w-3xl mx-auto mt-10 p-6 bg-white rounded shadow">
      <h2 className="text-2xl font-bold mb-4">ğŸ¯ Staff Redemption Panel</h2>
      {message && <p className="text-green-600 mb-4 font-semibold">{message}</p>}
      {deals.length === 0 ? (
        <p className="text-gray-600">No deals assigned to this location.</p>
      ) : (
        <ul className="space-y-3">
          {deals.map((deal) => (
            <li key={deal.id} className="border rounded p-4">
              <p className="font-semibold">{deal.title}</p>
              <p className="text-sm text-gray-600 mb-2">{deal.description}</p>
              <button
                className="px-4 py-2 bg-blue-600 text-white rounded hover:bg-blue-700"
                onClick={() => handleRedeem(deal.id)}
              >
                Redeem Deal
              </button>
            </li>
          ))}
        </ul>
      )}
    </div>
  );
}

export default StaffDashboard;
